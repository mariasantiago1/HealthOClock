CREATE TABLE TipoUso (
	`Id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`Tipo`	TEXT NOT NULL,
	`Uso`	TEXT NOT NULL
);
CREATE TABLE Remedio (
		`Id`	INTEGER NOT NULL,
		`Nome`	TEXT NOT NULL UNIQUE,
		PRIMARY KEY(`Id`)
	);
CREATE TABLE IF NOT EXISTS PlanoSaude (
	`Id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`Nome`	INTEGER NOT NULL
);
CREATE TABLE IF NOT EXISTS Paciente (
	`Id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`Nome`	TEXT NOT NULL,
	`DataNascimento`	NUMERIC NOT NULL,
	`Sexo`	TEXT NOT NULL CHECK(M),
	`PlanoSaude`	INTEGER,
	`TipoSanguineo`	TEXT
);
CREATE TABLE IF NOT EXISTS Login (
	`IdPaciente`	INTEGER NOT NULL,
	`Email`	TEXT NOT NULL UNIQUE,
	`Senha`	TEXT NOT NULL,
	PRIMARY KEY(`Id_paciente`),
	FOREIGN KEY(`Id_paciente`) REFERENCES `Paciente`(`Id`)
);
CREATE TABLE IF NOT EXISTS Exame (
	`IdANS`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`Descricao`	TEXT NOT NULL
);
CREATE TABLE IF NOT EXISTS DoencaCID (
	`IdCID`	TEXT NOT NULL,
	`Descricao`	TEXT NOT NULL,
	PRIMARY KEY(`Id_CID`)
);
CREATE TABLE IF NOT EXISTS ControleExame (
	`id`	INTEGER PRIMARY KEY AUTOINCREMENT,
	`nome_unidade`	    VARCHAR(100) NOT NULL,
	`endereco_unidade`	VARCHAR(100) NOT NULL,
	`nome_paciente`	    VARCHAR(100) NOT NULL,
	`dados_clinicos`	VARCHAR(100) NOT NULL,
	`especialidade_medico`	VARCHAR(100) NOT NULL,
	`material_examinar`	VARCHAR(100) NOT NULL,
	`tipo_exame`	    VARCHAR(100) NOT NULL,
	`data_realizacao`	VARCHAR(16) NOT NULL
);
